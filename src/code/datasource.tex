public DataSource createDataSource() {
  final OracleDataSource dataSource = new OracleDataSource();
  dataSource.setURL("jdbc:oracle:thin:@//192.168.0.16:1521/orcl");
  dataSource.setUser("User");
  dataSource.setPassword("Password");
  return dataSource;
}
